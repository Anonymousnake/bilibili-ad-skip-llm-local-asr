    const splitCaptionSections = text => {
        const lines = String(text || '').split('\n').filter(Boolean);
        const n = lines.length;
        if (!n) return { head: '', mid: '', tail: '' };
        return {
            head: lines.slice(0, Math.ceil(n * 0.3)).join(' '),
            mid: lines.slice(Math.ceil(n * 0.3), Math.ceil(n * 0.7)).join(' '),
            tail: lines.slice(Math.ceil(n * 0.7)).join(' ')

    const getRepeatedPhrases = (texts, minLen = 2, maxLen = 8) => {
        const map = new Map();
        texts.forEach(t => {
            const s = norm(t);
            for (let len = minLen; len <= maxLen; len++) {
                for (let i = 0; i <= s.length - len; i++) {
                    const sub = s.slice(i, i + len);
                    if (/^[，。！？、；：“”‘’（）【】\d]+$/.test(sub)) continue;
                    map.set(sub, (map.get(sub) || 0) + 1);
                }
            }
        });
        return [...map.entries()]
            .filter(([k, v]) => v >= 2 && k.length >= 2)
            .sort((a, b) => b[1] - a[1])
            .map(([k]) => k)
            .filter(k => !/的是了可以一个这个那个我们你们他们真的非常而且然后但是如果所以因为已经还是就是一下/.test(k))

        return arr.map(d => `[${secondsToTimeString(d.time)}] ${d.text}`).join('\n');
    };

    const findAdSegmentFromDanmaku = (arr, commentInfo) => {
    if (!arr.length) return { hit: false, score: 0, start: '无', end: '无', note: '弹幕为空' };

    const keys = extractCommentKeywords(commentInfo); // [{word, weight, type}, ...]
    const hits = arr.map(d => {
        const text = norm(d.text);
        const matched = keys.filter(k => text.includes(k.word));
        const weightSum = matched.reduce((s, x) => s + x.weight, 0);
        const strongCount = matched.filter(x => x.weight >= 3).length; // alias / phrase
        return { ...d, matched, weightSum, strongCount };
    }).filter(d => d.matched.length);

    if (hits.length < 3) {
        return { hit: false, score: 0, start: '无', end: '无', note: `命中相关弹幕过少(${hits.length}条)` };
    }

    let best = null;
    const W = 18;

    for (let i = 0; i < hits.length; i++) {
        const s = hits[i].time, e = s + W;
        const cluster = hits.filter(x => x.time >= s && x.time <= e);

        const totalWeight = cluster.reduce((sum, x) => sum + x.weightSum, 0);
        const strongDanmaku = cluster.filter(x => x.strongCount > 0).length;
        const uniqueWords = new Set(cluster.flatMap(x => x.matched.map(m => m.word))).size;
        const uniqueStrongWords = new Set(
            cluster.flatMap(x => x.matched.filter(m => m.weight >= 3).map(m => m.word))
        ).size;

        const score = totalWeight + strongDanmaku * 8 + uniqueWords * 2 + uniqueStrongWords * 6;

        if (!best || score > best.score) {
            best = { s, e, cluster, totalWeight, strongDanmaku, uniqueWords, uniqueStrongWords, score };
        }
    }

    if (!best) {
        return { hit: false, score: 0, start: '无', end: '无', note: '未形成有效弹幕聚集区' };
    }

    // 必须至少出现一个强词，否则泛词不可信
    if (best.strongDanmaku < 1 && best.uniqueStrongWords < 1) {
        return { hit: false, score: 0, start: '无', end: '无', note: '仅命中泛广告词，缺少平台/别名强锚点' };
    }

    let start = Math.max(0, best.cluster[0].time - 8);
    let end = best.cluster[best.cluster.length - 1].time + 20;
    if (end - start > 90) end = start + 90;

    return {
        hit: true,
        score: Math.min(95, 45 + Math.floor(best.score / 3)),
        start: secondsToTimeString(start),
        end: secondsToTimeString(end),
        note: `弹幕反查成功；强锚点弹幕${best.strongDanmaku}条，强关键词${best.uniqueStrongWords}类，总权重${best.totalWeight}`
    };
};

    const getCaptions = async (bvid, cid) => {
        try {
            const qs = await WbiUtils.encWbi({ bvid, cid });
            const qsStr = new URLSearchParams(qs).toString();
            const playerRes = await fetch(`https://api.bilibili.com/x/player/wbi/v2?${qsStr}`, { credentials: 'include' });
            const playerData = await playerRes.json();
            const subtitles = playerData.data?.subtitle?.subtitles || [];
            if (!subtitles.length) return "该视频未提供官方CC字幕。";

            const scoreSubtitle = sub => {
                const lang = `${sub?.lan || ''} ${sub?.lan_doc || ''}`.toLowerCase();
                let score = 0;
                if (/zh|cn|中文|汉语|国语/.test(lang)) score += 30;
                if (/人工|official|官方/.test(lang)) score += 10;
                if (/auto|自动|machine|机翻/.test(lang)) score -= 10;
                return score;
            };

            const candidates = [...subtitles].sort((a, b) => scoreSubtitle(b) - scoreSubtitle(a));
            for (const subtitle of candidates) {
                if (!subtitle?.subtitle_url) continue;
                const subUrl = /^https?:\/\//.test(subtitle.subtitle_url) ? subtitle.subtitle_url : `https:${subtitle.subtitle_url}`;
                try {
                    const subRes = await fetch(subUrl);
                    const subData = await subRes.json();
                    if (subData.body && subData.body.length) {
                        return subData.body.map(item => `[${secondsToTimeString(item.from)}] ${item.content}`).join('\n');
                    }
                } catch {}
            }

            return "字幕文件为空。";
        } catch {
            return "获取字幕过程中发生异常。";
        }
    };

    const formatTranscriptSegments = segments => {
        if (!Array.isArray(segments) || !segments.length) return '';
        return segments
            .map(seg => {
                const start = secondsToTimeString(Number(seg?.start) || 0);
                const text = String(seg?.text || '').trim();
                return text ? `[${start}] ${text}` : '';
            })

                settings.defaultSkip &&
                finalResult.start && finalResult.end &&
                finalResult.start !== '无' && finalResult.end !== '无'
            ) {
                const endTime = timeStringToSeconds(finalResult.end);
                clearSkipTimer();
                skipTimer = setInterval(() => {
                    const latestSettings = GM_getValue('ai_settings', DEFAULT_SETTINGS);
                    if (!latestSettings.defaultSkip) {
                        clearSkipTimer();
                        return;
                    }
                    const videoPlayer = document.querySelector('video');
                    if (!videoPlayer) return;
                    const rawStart = timeStringToSeconds(finalResult.start) || 0;
                    const startTime = rawStart > 15 ? rawStart - 6 : rawStart;
                    if (videoPlayer.currentTime >= startTime && videoPlayer.currentTime < endTime) {
                        showToast(`AI已为您跳过本段广告至 ${finalResult.end}`, 3000);
                        videoPlayer.currentTime = endTime;
                    }
                }, 500);
            } else if (
                finalResult.type === 'full_video_promo' &&
                finalResult.exist &&
                finalResult.finalProbability >= settings.probabilityThreshold
            ) {
                showToast('检测到整期推广视频：本次仅提示，不自动跳过。', 3500);
            }
        } catch (error) {
            console.error('分析脚本出错:', error);
            showToast(`识别失败: ${error.message}`, 4000);
        }
    };
